## Biimplication

```agda
⇔-equiv₁
  : ∀ {Γ} {φ ψ}
  → Γ ⊢ φ ⇔ ψ
  → Γ ⊢ (φ ⇒ ψ) ∧ (ψ ⇒ φ)

⇔-equiv₂
  : ∀ {Γ} {φ ψ}
  → Γ ⊢ (φ ⇒ ψ) ∧ (ψ ⇒ φ)
  → Γ ⊢ φ ⇔ ψ

⇔-equiv
  : ∀ {Γ} {φ ψ}
  → Γ ⊢ (φ ⇔ ψ) ⇔ ((φ ⇒ ψ) ∧ (ψ ⇒ φ))

⇔-assoc₁
  : ∀ {Γ} {φ ψ γ}
  → Γ ⊢ φ ⇔ (ψ ⇔ γ)
  → Γ ⊢ (φ ⇔ ψ) ⇔ γ

⇔-assoc₂
  : ∀ {Γ} {φ ψ γ}
  → Γ ⊢ (φ ⇔ ψ) ⇔ γ
  → Γ ⊢ φ ⇔ (ψ ⇔ γ)

⇔-assoc
  : ∀ {Γ} {φ ψ γ}
  → Γ ⊢ (φ ⇔ (ψ ⇔ γ)) ⇔ ((φ ⇔ ψ) ⇔ γ)

⇔-comm
  : ∀ {Γ} {φ ψ}
  → Γ ⊢ φ ⇔ ψ
  → Γ ⊢ ψ ⇔ φ

⇒-⇔-¬∨
  : ∀ {Γ} {φ ψ}
  → Γ ⊢ (φ ⇒ ψ) ⇔ (¬ φ ∨ ψ)

thm-bicon₀
  : ∀ {Γ} {φ ψ}
  → Γ ⊢ φ ⇔ ψ → Γ ⊢ ¬ φ
  → Γ ⊢ ¬ ψ
⇔-¬-to-¬ = thm-bicon₀

thm-bicon₁
  : ∀ {Γ} {φ ψ}
  → Γ ⊢ ¬ φ ⇔ ψ → Γ ⊢ φ
  → Γ ⊢ ¬ ψ
¬⇔-to-¬ = thm-bicon₁

¬-equiv
  : ∀ {Γ} {φ}
  → Γ ⊢ ¬ φ ⇔ (φ ⇒ ⊥)

¬¬-equiv
  : ∀ {Γ} {φ}
  → Γ ⊢ ¬ (¬ φ) ⇔ φ

⇒⇒-⇔-∧⇒
  : ∀ {Γ} {φ ψ γ}
  → Γ ⊢ (φ ⇒ (ψ ⇒ γ)) ⇔ ((φ ∧ ψ) ⇒ γ)

⇔-trans
  : ∀ {Γ} {φ ψ γ}
  → Γ ⊢ γ ⇔ φ
  → Γ ⊢ φ ⇔ ψ
  → Γ ⊢ γ ⇔ ψ
subst⊢⇔₁ = ⇔-trans

```

∨-assoc₂

∨-dist₁

∨-dist₂
  : ∀ {Γ} {φ ψ γ}

∨-dist

∨-to-¬¬∧¬ = ∨-equiv

∨-dmorgan

  → Γ ⊢ ψ ∨ γ

resolve₁

resolve₇

  : ∀ {Γ} {φ ψ}

contraposition₁ = vanDalen244d

contraposition₂

## Negation

```agda

¬-equiv₁
  : ∀ {Γ} {φ}
  → Γ ⊢ ¬ φ
  → Γ ⊢ φ ⇒ ⊥
¬-to-⇒⊥ = ¬-equiv₁

¬-equiv₂
  : ∀ {Γ} {φ}
  → Γ ⊢ φ ⇒ ⊥
  → Γ ⊢ ¬ φ
⇒⊥-to-¬ = ¬-equiv₂

¬¬-equiv₁
  : ∀ {Γ} {φ}
  → Γ ⊢ ¬ (¬ φ)
  → Γ ⊢ φ

¬¬-equiv₂
  : ∀ {Γ} {φ}
  → Γ ⊢ φ
  → Γ ⊢ ¬ (¬ φ)

¬∨-to-⇒
  : ∀ {Γ} {φ ψ}
  → Γ ⊢ ¬ φ ∨ ψ
  → Γ ⊢ φ ⇒ ψ

¬⊤-to-⊥
  : ∀ {Γ}
  → Γ ⊢ ¬ ⊤
  → Γ ⊢ ⊥

⊤-to-¬⊥
  : ∀ {Γ}
  → Γ ⊢ ⊤
  → Γ ⊢ ¬ ⊥

¬⊥-to-⊤
  : ∀ {Γ}
  → Γ ⊢ ¬ ⊥
  → Γ ⊢ ⊤

⊥-to-¬⊤
  : ∀ {Γ}
  → Γ ⊢ ⊥
  → Γ ⊢ ¬ ⊤

¬-inside
  : ∀ {Γ} {φ ψ}
  → φ ≡ ψ
  → Γ ⊢ ¬ φ
  → Γ ⊢ ¬ ψ
≡-¬-to-¬ = ¬-inside

subst⊢¬
  : ∀ {Γ} {φ γ}
  → Γ ⊢ γ ⇒ φ
  → Γ ⊢ ¬ φ
  → Γ ⊢ ¬ γ

¬⇔-to-⇒¬∧⇒¬
  : ∀ {Γ} {φ₁ φ₂}
  → Γ ⊢ ¬ (φ₁ ⇔ φ₂)
  → Γ ⊢ (φ₁ ⇒ ¬ φ₂) ∧ (φ₂ ⇒ ¬ φ₁)

```
